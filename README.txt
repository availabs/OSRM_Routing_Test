"npm install" = install node modules
"setup_arizona.sh" = download and process OSM data for all of Arizona into OSRM data
"python tile_download.py --config=your_config_file" = downloads height maps as .png files
	use -h for help
"sudo apt-get install libsdl2-2.0 libsdl2-dev libsdl2-image-dev" = instal SDl2 , SDL2 dev and  SDL2_image libraries
"make" = builds the c++ stuff into out file "main"
	requires lib-png 1.6
"./tile_process folder out_file" = reads tiles from "folder" and ouputs an OSM XML file to "out_file"
"process_osm.sh folder file" = process the OSM "file" and output to data/"folder"
"npm start" = starts the test map web server
